<script>
const introText = "안녕하세요! 👋 wdohoon입니다!";
let index = 0;
function typeEffect() {
    if (index < introText.length) {
        document.getElementById("animated-intro").innerHTML += introText.charAt(index);
        index++;
        setTimeout(typeEffect, 100);
    }
}
window.onload = typeEffect;
</script>
